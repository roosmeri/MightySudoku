4   package fi.mightysudoku.logiikka;
5   
6   import java.util.ArrayList;
7   
8   /**
9    * Luokka edustaa Sarake-olioita, jotka koostuvaat Ruuduista.
10   *
11   *
12   */
13  public class Sarake implements Numeromuodostelma {
14  
15      private ArrayList<Ruutu> ruudut = new ArrayList<>();
16      private int saraketunnus = 0;
17  
18      /**
19       * Konstruktori asettaa Sarakkeen muodostavat Ruudut talteen ja muodostaa
20       * saraketunnuksen.
21       *
22       * @param ruudut Pelialustan määräämät ruudut
23       */
24      public Sarake(ArrayList<Ruutu> ruudut) {
25          this.ruudut = ruudut;
26          this.saraketunnus = this.ruudut.get(0).getX();
27      }
28  
29      /**
30       * Metodi tarkistaa onko sarakkeessa ehdotettu arvo.
31       *
32       * @param arvo Ehdotettu arvo.
33       *
34       * @return True jos on ja false jos ei ole.
35       */
36      @Override
37      public boolean onkoArvoJo(int arvo) {
38          //tarkistetaan onko rivillä jo ehdotettu arvo
39          for (Ruutu r : this.ruudut) {
40              if (r.getArvo() == arvo) {
41                  return true;
42              }
43          }
44          return false;
45      }
46  
47      /**
48       * Metodi palauttaa sarakkeeseen kuuluvat ruudut.
49       *
50       * @return Ruudut.
51       */
52      @Override
53      public ArrayList<Ruutu> getRuudut() {
54          return this.ruudut;
55      }
56  
57      /**
58       * Metodi palauttaa sarakkeen tunnuksen.
59       *
60       * @return Saraketunnus.
61       */
62      public int getSaraketunnus() {
63          return saraketunnus;
64      }
65  
66  }
