5   /**
6    * Luokka edustaa Sudokuruudukon aliruudukkoa 3x3 kokoista ruudukkoa.
7    *
8    */
9   public class Ruudukko implements Numeromuodostelma {
10  
11      private ArrayList<Ruutu> ruudut = new ArrayList<>();
12      private String ruudukontunnus = "";
13  
14      /**
15       * Konstruktori asettaa ruudukkoon siihen kuuluvat ruudut.
16       *
17       * @param ruudut Ruudukon ruudut.
18       */
19      public Ruudukko(ArrayList<Ruutu> ruudut) {
20          this.ruudut = ruudut;
21          this.ruudukontunnus = "" + ruudut.get(0).getX() + ruudut.get(0).getY();
22      }
23  
24      /**
25       * Metodi tarkistaa onko ruudukossa jo asetettavaa arvoa.
26       *
27       * @param arvo.
28       *
29       * @return True jos on ja false jos ei ole TAI sama arvo on siin√§ ruudussa
30       * mihina asetetaan.
31       */
32      @Override
33      public boolean onkoArvoJo(Ruutu ruutu, int arvo) {
34  
35          for (Ruutu r : this.ruudut) {
36              if (r.getArvo() == arvo) {
37                  if (r.getX() == ruutu.getX() && r.getY() == ruutu.getY()) {
38                      return false;
39                  } else {
40                      return true;
41                  }
42              }
43          }
44          return false;
45      }
46  
47      @Override
48      public ArrayList<Ruutu> getRuudut() {
49          return this.ruudut;
50      }
51  
52      public String getRuudukontunnus() {
53          return ruudukontunnus;
54      }
55  
56  }
