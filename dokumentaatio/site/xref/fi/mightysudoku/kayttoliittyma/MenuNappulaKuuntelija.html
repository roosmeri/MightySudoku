1   package fi.mightysudoku.kayttoliittyma;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   import javax.swing.JOptionPane;
7   
8   /**
9    * Kuuntelija SudokuFramen menunappuloille.
10   *
11   * @author Viliina
12   */
13  public class MenuNappulaKuuntelija implements ActionListener {
14  
15      SudokuFrame sudoku;
16  
17      /**
18       * Konstruktori asettaa SudokuFramen kuuntelijalle.
19       *
20       * @param sudokuframe SudokuFrame.
21       */
22      public MenuNappulaKuuntelija(SudokuFrame sudokuframe) {
23          this.sudoku = sudokuframe;
24      }
25  
26      @Override
27      public void actionPerformed(ActionEvent e) {
28          JButton painettu = (JButton) e.getSource();
29          if (painettu.getText().equals("Lopeta")) {
30              Object[] vaihtoehdot = {"Haluan", "En"};
31              int varmistus = JOptionPane.showOptionDialog(sudoku, "Haluatko varmasti lopettaa?", "Lopeta Sudokun pelaaminen", JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null, vaihtoehdot, vaihtoehdot[0]);
32              if (varmistus == JOptionPane.YES_OPTION) {
33                  System.exit(0);
34              }
35              System.exit(0);
36          } else if (painettu.getText().equals("Aloita alusta")) {
37              Object[] vaihtoehdot = {"Haluan", "En"};
38              int varmistus = JOptionPane.showOptionDialog(sudoku, "Haluatko varmasti aloittaa alusta?", "Aloita tämä peli alusta", JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null, vaihtoehdot, vaihtoehdot[0]);
39              if (varmistus == JOptionPane.YES_OPTION) {
40                  sudoku.tyhjennaTaytetyt();
41              }
42          } else if (painettu.getText().equals("Uusi peli")) {
43              Object[] vaihtoehdot = {"Helppo", "Keskitaso", "Vaikea"};
44              String taso = (String) JOptionPane.showInputDialog(sudoku, "Minkä tasoista sudokua haluat pelata?", "Uusi peli", JOptionPane.PLAIN_MESSAGE, null, vaihtoehdot, "Helppo");
45              if (taso != null && taso.length() > 0) {
46                  if (taso.equals("Helppo")) {
47                      sudoku.asetanumerot(1);
48                  } else if (taso.equals("Keskitaso")) {
49                      sudoku.asetanumerot(2);
50                  } else if (taso.equals("Vaikea")) {
51                      sudoku.asetanumerot(3);
52                  }
53              } else {
54                  return;
55              }
56          }
57      }
58  
59  }
