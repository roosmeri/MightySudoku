22  /**
23   * Luokka on JFramen aliluokka, joka muodostaa visuaalisen esityksen pelistä.
24   *
25   * @author Viliina
26   */
27  public class SudokuFrame extends JFrame {
28  
29      private Pelialusta alusta = new Pelialusta();
30      private SudokuGeneraattori generaattori = new SudokuGeneraattori(alusta);
31      private JFormattedTextField[][] alustanruudut;
32      private final int ruudunkoko = 60;
33      private final int korkeus = ruudunkoko * 9;
34      private final int leveys = ruudunkoko * 9;
35      private final Font fontti = new Font("Monospaced", Font.BOLD, 25);
36      private int[][] numerot = new int[9][9];
37      private ActionListener numerokuuntelija = new NumeronAsetusKuuntelija(this);
38      private NumberFormatter vainNumerot = new NumberFormatter(NumberFormat.getInstance());
39  
40      /**
41       * Konstruktori asettaa otsikon ja asettaa numerot kohdilleen
42       * sudokuruudukkoon.
43       *
44       * @param otsikko On ikkunan otsikko.
45       * @throws HeadlessException .
46       */
47      public SudokuFrame(String otsikko) throws HeadlessException {
48          super(otsikko);
49          rajoitaSyotteenMuotoilua();
50          asetanumerot();
51          this.alustanruudut = new JFormattedTextField[9][9];
52          sisallonAsetus();
53  
54      }
55  
56      /**
57       * Metodi määrittelee formatteria syotteelle.
58       */
59  
60      public void rajoitaSyotteenMuotoilua() {
61          vainNumerot.setValueClass(Integer.class);
62          vainNumerot.setMinimum(1);
63          vainNumerot.setMaximum(9);
64      }
65  
66      /**
67       * Metodi määrittelee Container-olion, asettaa JTextField-ruudut
68       * muokattavaksi, jos niissä ei ollut valmiina asetettua arvoa. Metodi myös
69       * määrittelee ruudukon ja ruutujen ulkomuodon.
70       *
71       */
72      public void sisallonAsetus() {
73          Container container = getContentPane();
74          container.setLayout(new GridLayout(9, 9));
75          for (int rivi = 0; rivi < 9; ++rivi) {
76              for (int sarake = 0; sarake < 9; ++sarake) {
77                  alustanruudut[rivi][sarake] = new JFormattedTextField(vainNumerot);
78                  container.add(alustanruudut[rivi][sarake]);
79                  if (numerot[rivi][sarake] == (0)) {
80                      alustanruudut[rivi][sarake].setText("");
81                      alustanruudut[rivi][sarake].setEditable(true);
82                      alustanruudut[rivi][sarake].setBackground(Color.WHITE);
83  
84                      alustanruudut[rivi][sarake].addActionListener(numerokuuntelija);
85                  } else {
86                      alustanruudut[rivi][sarake].setText(numerot[rivi][sarake] + "");
87                      alustanruudut[rivi][sarake].setEditable(false);
88                      alustanruudut[rivi][sarake].setBackground(Color.LIGHT_GRAY);
89                      alustanruudut[rivi][sarake].setForeground(Color.BLACK);
90                  }
91                  alustanruudut[rivi][sarake].setHorizontalAlignment(JTextField.CENTER);
92                  alustanruudut[rivi][sarake].setFont(fontti);
93              }
94          }
95  
96          container.setPreferredSize(new Dimension(korkeus, leveys));
97          pack();
98  
99          setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
100         setVisible(true);
101     }
102 
103     /**
104      * Metodi kutsuu asettaa ensin Pelialustaksi SudokuGeneraattorin antaman
105      * alustan. Metodi asettaa Ruutu olioiden arvot niitä vastaaviin
106      * JTextField-olioihin.
107      */
108     public void asetanumerot() {
109         this.alusta = generaattori.generoiPelialusta();
110         for (Ruutu ruutu : alusta.getRuudut()) {
111             numerot[ruutu.getX()][ruutu.getY()] = ruutu.getArvo();
112         }
113     }
114 
115     public Pelialusta getAlusta() {
116         return alusta;
117     }
118 
119     /**
120      * Metodi palauttaa JFormattedTextField:in joka sijaitsee kohdassa x,y.
121      *
122      * @param x x-koordinaatti.
123      * @param y y-koordinaatti.
124      * @return JFormattedTextField annetussa sijainnissa.
125      */
126     public JFormattedTextField getTextFieldAt(int x, int y) {
127         return alustanruudut[x][y];
128     }
129 
130 }
