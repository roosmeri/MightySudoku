46      /**
47       * Konstruktori asettaa otsikon ja asettaa numerot kohdilleen
48       * sudokuruudukkoon ja kysyy sitten Kayttajalta vaikeustasoa jonka
49       * perusteella valmiit numerot asetetaan ruudukkoon.
50       *
51       * @param otsikko On ikkunan otsikko.
52       * @throws HeadlessException .
53       */
54      public SudokuFrame(String otsikko) throws HeadlessException {
55          super(otsikko);
56          this.alustanruudut = new JTextField[9][9];
57          sisallonAsetus();
58          Object[] vaihtoehdot = {"Helppo", "Keskitaso", "Vaikea"};
59          String taso = (String) JOptionPane.showInputDialog(this, "Minkä tasoista sudokua haluat pelata?", "Uusi peli", JOptionPane.PLAIN_MESSAGE, null, vaihtoehdot, "Helppo");
60          if (taso != null && taso.length() > 0) {
61              if (taso.equals("Helppo")) {
62                  asetanumerot(1);
63              } else if (taso.equals("Keskitaso")) {
64                  asetanumerot(2);
65              } else if (taso.equals("Vaikea")) {
66                  asetanumerot(3);
67              }
68          } else {
69              return;
70          }
71      }

111     private void luoRuudukko(Container container) {
112         JPanel sudokuruudukko = new JPanel();
113         sudokuruudukko.setLayout(new GridLayout(9, 9));
114         sudokuruudukko.setPreferredSize(new Dimension(korkeus, leveys));
115         container.add(sudokuruudukko);
116         ActionListener syotekuuntelija = new SyoteKuuntelija(this);
117         for (int rivi = 0; rivi < 9; ++rivi) {
118             for (int sarake = 0; sarake < 9; ++sarake) {
119                 alustanruudut[rivi][sarake] = new JTextField();
120                 asetaReunat(rivi, sarake);
121                 sudokuruudukko.add(alustanruudut[rivi][sarake]);
122                 if (numerot[rivi][sarake] == (0)) {
123                     alustanruudut[rivi][sarake].setText("");
124                     alustanruudut[rivi][sarake].setEditable(true);
125                     alustanruudut[rivi][sarake].setBackground(Color.WHITE);
126                     alustanruudut[rivi][sarake].addActionListener(syotekuuntelija);
127                 } else {
128                     alustanruudut[rivi][sarake].setText(numerot[rivi][sarake] + "");
129                     alustanruudut[rivi][sarake].setEditable(false);
130                     alustanruudut[rivi][sarake].setBackground(Color.LIGHT_GRAY);
131                     alustanruudut[rivi][sarake].setForeground(Color.BLACK);
132                 }
133                 alustanruudut[rivi][sarake].setHorizontalAlignment(JTextField.CENTER);
134                 alustanruudut[rivi][sarake].setFont(fontti);
135             }
136         }
137     }

139     private void asetaReunat(int rivi, int sarake) {
140         alustanruudut[rivi][sarake].setBorder(BorderFactory.createMatteBorder(1, 1, 1, 1, Color.BLACK));
141         if (rivi == 2 || rivi == 5) {
142             alustanruudut[rivi][sarake].setBorder(BorderFactory.createMatteBorder(1, 1, 3, 1, Color.BLACK));
143         }
144         if (sarake == 2 || sarake == 5) {
145             if (rivi == 2 || rivi == 5) {
146                 alustanruudut[rivi][sarake].setBorder(BorderFactory.createMatteBorder(1, 1, 3, 3, Color.BLACK));
147             } else {
148                 alustanruudut[rivi][sarake].setBorder(BorderFactory.createMatteBorder(1, 1, 1, 3, Color.BLACK));
149             }
150         }
151     }

153     /**
154      * Metodi käyttää SudokuGeneraattoria uuden pelialustan luomiseen
155      * vaikeustason mukaan. Asettaa ruutujen arvot numeroruudukkoon.
156      *
157      * @param taso Vaikeustaso.
158      */
159     public void asetanumerot(int taso) {
160         tyhjenna.setEnabled(true);
161         generoiPelialusta(taso);
162         this.alusta = generaattori.getAlusta();
163         for (Ruutu ruutu : alusta.getRuudut()) {
164             numerot[ruutu.getX()][ruutu.getY()] = ruutu.getArvo();
165         }
166         for (int rivi = 0; rivi < 9; ++rivi) {
167             for (int sarake = 0; sarake < 9; ++sarake) {
168                 if (numerot[rivi][sarake] == (0)) {
169                     alustanruudut[rivi][sarake].setText("");
170                     alustanruudut[rivi][sarake].setEditable(true);
171                     alustanruudut[rivi][sarake].setBackground(Color.WHITE);
172                 } else {
173                     alustanruudut[rivi][sarake].setText(numerot[rivi][sarake] + "");
174                     alustanruudut[rivi][sarake].setEditable(false);
175                     alustanruudut[rivi][sarake].setBackground(Color.LIGHT_GRAY);
176                     alustanruudut[rivi][sarake].setForeground(Color.BLACK);
177                 }
178             }
179 
180         }
181     }

183     private void generoiPelialusta(int taso) {
184         Random r = new Random();
185         if (taso == 1) {
186             int tyhjia = r.nextInt((49 - 18) + 1) + 18;
187             this.generaattori = new SudokuGeneraattori(alusta, tyhjia);
188         } else if (taso == 2) {
189             int tyhjia = r.nextInt((54 - 50) + 1) + 50;
190             this.generaattori = new SudokuGeneraattori(alusta, tyhjia);
191         } else if (taso == 3) {
192             int tyhjia = r.nextInt((60 - 51) + 1) + 51;
193             this.generaattori = new SudokuGeneraattori(alusta, tyhjia);
194         }
195     }

212     /**
213      * Metodi tarkastamaan onko sudoku ratkaistu numeron asetuksen jälkeen.
214      *
215      * @return true jos yhtään puna- tai valkotaustaista ruutua ei löytnyt,
216      * muuten false.
217      */
218     public boolean onkoRatkaistu() {
219         for (int i = 0; i < alustanruudut.length; i++) {
220             for (int j = 0; j < alustanruudut.length; j++) {
221                 if (getTextFieldAt(i, j).getBackground() == Color.red
222                         || getTextFieldAt(i, j).getBackground() == Color.WHITE) {
223                     return false;
224                 }
225             }
226         }
227         lukitseKaikkiRuudut();
228         tyhjenna.setEnabled(false);
229         return true;
230 
231     }
232 
233     private void lukitseKaikkiRuudut() {
234         for (int i = 0; i < alustanruudut.length; i++) {
235             for (int j = 0; j < alustanruudut.length; j++) {
236                 getTextFieldAt(i, j).setEditable(false);
237             }
238         }
239     }

241     /**
242      * Metodi tyhjentää kaikki käyttäjän täytettävissä olleet ruudut.
243      */
244     public void tyhjennaTaytetyt() {
245         for (int i = 0; i < alustanruudut.length; i++) {
246             for (int j = 0; j < alustanruudut.length; j++) {
247                 if (getTextFieldAt(i, j).isEditable()) {
248                     getTextFieldAt(i, j).setText("");
249                     getTextFieldAt(i, j).setBackground(Color.WHITE);
250                     alusta.haeRuutu(i, j).setArvo(0);
251                 }
252             }
253 
254         }
255     }
